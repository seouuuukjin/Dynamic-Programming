11053 - 가장 긴 증가하는 부분수열

https://www.acmicpc.net/problem/11053

```c++
//lis = longest increasing subsequece
#include <iostream>
#include <algorithm>
using namespace std;
int lis_len[1111] = {1,};  // 주의~!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
//배열초기화 방법 오류
int arr[1111] = {0};

int main() {
	int n;
	scanf("%d", &n);
	for(int i=0; i<n; i++)
		scanf("%d", &arr[i]);
	for(int i=0; i<n; i++)
		printf("%d ", lis_len[i]);
	cout << endl;
	for(int i =1; i<=n; i++){ // i가 0일때는 무조건 lis_len[i] = 1이다.
		for(int j = i-1; j>=0; j--){
			if(arr[i] > arr[j])
				lis_len[i] = max(lis_len[j]+1, lis_len[i]);
		}
	}
	for(int i=0; i<n; i++)
		printf("%d ", lis_len[i]);
	cout << endl;
	printf("%d", *max_element(lis_len, lis_len+n));
	return 0;
}
```
